DROP TABLE IF EXISTS subforums;
CREATE TABLE IF NOT EXISTS subforums (
	subforum_id tinyint unsigned primary key auto_increment,
	subforum_name varchar(100),
	subforum_order tinyint unsigned
);

DROP TABLE IF EXISTS boards;
CREATE TABLE IF NOT EXISTS boards (
	board_id tinyint unsigned primary key auto_increment,
	subforum_id tinyint unsigned,
	board_name varchar(100),
	board_description text,
	board_order tinyint unsigned,
	last_post_id mediumint unsigned NULL,
	num_threads mediumint unsigned default '0',
	num_posts mediumint unsigned default '0',
	auth_view tinyint unsigned default '5',
	anonymous_posting enum('true','false') default 'false'
);

DROP TABLE IF EXISTS threads;
CREATE TABLE IF NOT EXISTS threads (
	thread_id mediumint unsigned primary key auto_increment,
	board_id tinyint unsigned,
	thread_name varchar(100),
	poster_id mediumint unsigned,
	thread_text text,
	time_posted datetime,
	time_updated datetime,
	num_replies smallint unsigned default '0',
	locked enum('true','false') default 'false',
	sticky enum('true','false') default 'false',
	deleted enum('true','false') default 'false'
);

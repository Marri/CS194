CREATE TABLE subforums (
	subforum_id tinyint unsigned primary key auto_increment,
	subforum_name varchar(100),
	subforum_order tinyint unsigned
);

CREATE TABLE boards (
	board_id tinyint unsigned primary key auto_increment,
	subforum_id tinyint unsigned,
	board_name varchar(100),
	board_description text,
	board_order tinyint unsigned,
	last_post_id mediumint unsigned NULL,
	num_threads mediumint unsigned default '0',
	num_posts mediumint unsigned default '0',
	auth_view tinyint unsigned default '5', //What user level can view this forum, default everyone
	anonymous_posting enum('true','false') default 'false'
);

CREATE TABLE threads (
	thread_id mediumint unsigned primary key auto_increment,
	board_id tinyint unsigned,
	thread_name varchar(100),
	poster_id mediumint unsigned,
	thread_text text,
	time_posted datetime,
	time_updated datetime,
	num_replies smallint unsigned default '0',
	locked enum('true','false') default 'false',
	sticky enum('true','false') default 'false',
	deleted enum('true','false') default 'false'
);

CREATE TABLE posts (
	post_id mediumint unsigned primary key auto_increment,
	thread_id mediumint unsigned,
	board_id tinyint unsigned,
	poster_id mediumint unsigned,
	poster_name varchar(100),
	post_time datetime,
	edit_time datetime
);

CREATE TABLE post_text (
	post_id mediumint unsigned,
	post_text text
);

CREATE TABLE squffy_items (

	item_id smallint unsigned,

	item_order tinyint unsigned default '0'

CREATE TABLE squffy_family (
	squffy_id int unsigned,

CREATE TABLE squffy_appearance (

CREATE TABLE appearance_traits (
	trait_id tinyint unsigned primary keyauto_increment,
	trait_name varchar(255),
	trait_title varchar(255),
	trait_type enum('1','2') default '1',
	design_activated enum('true','false') default 'false',
	activated enum('true','false') default 'false'
);

CREATE TABLE personality_traits (
	trait_id tinyint unsigned primary keyauto_increment,
	trait_good_name varchar(100),
	trait_bad_name varchar(100),
	trait_good_desc text,
	trait_bad_desc text,

CREATE TABLE species (
	species_id tinyint unsigned auto_increment,
	species_name varchar(255),
	design_activated enum('true','false') default 'false',
	activated enum('true','false') default 'false',
	primary key (speciesid)

CREATE TABLE layouts (

CREATE TABLE messages (
	message_id int unsigned primary key auto_increment,
	to_id mediumint unsigned,
	from_id mediumint unsigned,
	message_type tinyint unsigned,
	subject varchar(127),
	message text,
	time_sent datetime,
	is_read enum('true','false') default 'false',
	in_inbox enum('true','false') default 'true',
	in_outbox enum('true','false') default 'true'
);

CREATE TABLE news (
	news_id mediumint unsigned primary key auto_increment,

	content text,
	date_posted datetime,
	poster_id mediumint unsigned

CREATE TABLE news_comments (
	comment_id mediumint unsigned primary key auto_increment,
	news_id mediumint unsigned,

	content text,
	date_posted datetime
